## 2. 技术趋势

### 城市自动驾驶

```cpp
// 城市NOA架构
class UrbanNOA {
public:
    void navigate(Point destination) {
        // 1. 路径规划
        auto route = planner.plan(ego_position, destination);
        
        // 2. 行为决策
        for (auto& behavior : route.behaviors) {
            auto prediction = predictor.predict(behavior);
            
            // 3. 运动规划
            auto trajectory = planner.plan_motion(
                ego_state, 
                prediction,
                constraints
            );
            
            // 4. 执行控制
            controller.execute(trajectory);
        }
    }
};
```
